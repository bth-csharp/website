import {{ useState }} from 'react';

# ✅ Installationscheck

Innan du börjar med kursen, se till att du har installerat följande verktyg:

<InstallChecklist />

export function InstallChecklist() {
  const [git, setGit] = useState(false);
  const [dotnet, setDotnet] = useState(false);

  const allDone = git && dotnet;

  return (
    <div>
      <label>
        <input type="checkbox" checked={git} onChange={() => setGit(!git)} />
        Jag har installerat Git
      </label>
      <br />
      <label>
        <input type="checkbox" checked={dotnet} onChange={() => setDotnet(!dotnet)} />
        Jag har installerat C# / .NET SDK
      </label>
      <br />
      {allDone && <p>✅ Du är redo att köra igång!</p>}
    </div>
  );
}
