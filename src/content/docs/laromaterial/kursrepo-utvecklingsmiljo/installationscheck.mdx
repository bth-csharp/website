import {{ useState }} from 'react'

export default function InstallCheck() {
  const [git, setGit] = useState(false)
  const [dotnet, setDotnet] = useState(false)

  return (
    <div>
      <h1>✅ Installationscheck</h1>
      <p>Innan du börjar kursen, se till att du har installerat följande verktyg:</p>

      <label>
        <input type="checkbox" checked={git} onChange={() => setGit(!git)} />
        Jag har installerat <strong>Git</strong>
      </label>
      <br />

      <label>
        <input type="checkbox" checked={dotnet} onChange={() => setDotnet(!dotnet)} />
        Jag har installerat <strong>C# / .NET SDK</strong>
      </label>

      <p style={{ marginTop: '1em' }}>
        {git && dotnet ? '✅ Du är redo att köra igång!' : '⬅️ Bocka av båda för att fortsätta'}
      </p>
    </div>
  )
}
